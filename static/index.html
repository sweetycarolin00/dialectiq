function startListening(){ if(!SR){ $('statusText').textContent = 'Mic not supported here'; $('statusSub').textContent = 'Try Chrome or Edge.'; return; } // reset finalTranscript = ''; recognition = new SR(); recognition.lang = state.from.code; // prevent repeated listening recognition.continuous = false; // only final speech recognition.interimResults = false; let translated = false; recognition.onresult = (ev)=>{ const result = ev.results[ ev.results.length - 1 ]; if( result && result.isFinal ){ const transcript = result[0] .transcript .trim(); if( transcript && transcript !== finalTranscript ){ finalTranscript = transcript; $('statusSub').textContent = transcript; } } }; recognition.onerror = (ev)=>{ recognizing = false; $('micWrap') .classList .remove( 'listening' ); $('micBtn') .classList .remove( 'stop-mode' ); $('statusText') .textContent = 'Could not hear you'; $('statusSub') .textContent = ev.error === 'not-allowed' ? 'Microphone blocked.' : 'Try again.'; }; recognition.onend = ()=>{ recognizing = false; $('micWrap') .classList .remove( 'listening' ); $('micBtn') .classList .remove( 'stop-mode' ); const text = finalTranscript .trim(); if( text && !translated ){ translated = true; $('statusText') .textContent = 'Translating...'; setTimeout(()=>{ processTranslation( text ); }, 600); finalTranscript = ''; }else{ $('statusText') .textContent = 'Tap the mic to speak'; $('statusSub') .textContent = 'Try again.'; } }; recognition.start(); recognizing = true; $('micWrap') .classList .add( 'listening' ); $('micBtn') .classList .add( 'stop-mode' ); $('statusText') .textContent = 'Listening...'; $('statusSub') .textContent = 'Speak now.'; }
