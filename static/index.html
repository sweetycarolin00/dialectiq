function startListening(){ if(!SR){ $('statusText').textContent='Mic not supported here'; $('statusSub').textContent='Try Chrome or Edge.'; return; } finalTranscript=''; recognition = new SR(); recognition.lang = state.from.code; // Stop repeated sessions recognition.continuous = false; // Only final text recognition.interimResults = false; recognition.onresult = (ev)=>{ const result = ev.results[ ev.results.length - 1 ]; if( result && result.isFinal ){ finalTranscript = result[0] .transcript .trim(); $('statusSub').textContent = finalTranscript; } }; recognition.onerror = (ev)=>{ stopListening(); $('statusText').textContent = 'Could not hear you'; $('statusSub').textContent = ev.error === 'not-allowed' ? 'Microphone blocked.' : 'Try again.'; }; recognition.onend = ()=>{ recognizing = false; $('micWrap') .classList .remove('listening'); $('micBtn') .classList .remove('stop-mode'); const text = finalTranscript .trim(); if(text){ // prevent duplicate call finalTranscript=''; processTranslation(text); }else{ $('statusText').textContent = 'Tap the mic to speak'; $('statusSub').textContent = 'Try again.'; } }; recognition.start(); recognizing = true; $('micWrap') .classList .add('listening'); $('micBtn') .classList .add('stop-mode'); $('statusText').textContent = 'Listening...'; $('statusSub').textContent = 'Speak now.'; }
