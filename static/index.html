function startListening(){ if(!SR){ $('statusText').textContent='Mic not supported'; $('statusSub').textContent='Use Chrome or Edge'; return; } finalTranscript=''; recognition = new SR(); recognition.lang = state.from.code; recognition.continuous = false; recognition.interimResults = false; recognition.onresult = (ev)=>{ if( ev.results && ev.results.length ){ finalTranscript = ev.results[0][0] .transcript .trim(); $('statusSub').textContent = finalTranscript; } }; recognition.onerror = ()=>{ recognizing=false; $('micWrap') .classList .remove('listening'); $('micBtn') .classList .remove('stop-mode'); $('statusText').textContent = 'Try again'; }; recognition.onend = async ()=>{ recognizing=false; $('micWrap') .classList .remove('listening'); $('micBtn') .classList .remove('stop-mode'); const text = finalTranscript; finalTranscript=''; if(text){ $('statusText').textContent = 'Translating...'; await processTranslation( text ); }else{ $('statusText').textContent = 'Tap mic'; $('statusSub').textContent = ''; } }; recognition.start(); recognizing=true; $('micWrap') .classList .add('listening'); $('micBtn') .classList .add('stop-mode'); $('statusText').textContent = 'Listening...'; $('statusSub').textContent = ''; }
