recognition.onresult = (ev) => { let interim = ""; // Only process NEW results for (let i = ev.resultIndex; i < ev.results.length; i++) { const result = ev.results[i]; const transcript = result[0].transcript.trim(); if (!transcript) continue; if (result.isFinal) { // Prevent duplicate final chunks const last = finalChunks[finalChunks.length - 1]; if ( !last || ( transcript !== last && !transcript.includes(last) && !last.includes(transcript) ) ) { finalChunks.push(transcript); } else { // Replace shorter duplicate with newer longer phrase finalChunks[finalChunks.length - 1] = transcript.length > last.length ? transcript : last; } } else { interim += transcript + " "; } } finalTranscript = finalChunks.join(" ").trim(); $('statusSub').textContent = (finalTranscript + " " + interim).trim() || "Listening..."; };
