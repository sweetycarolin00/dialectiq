function startListening() { if (!SR) { $('statusText').textContent = 'Mic not supported'; return; } finalTranscript = ''; recognition = new SR(); recognition.lang = state.from.code; recognition.continuous = false; recognition.interimResults = false; recognition.onresult = (ev) => { const transcript = ev.results[0][0] .transcript .trim(); finalTranscript = transcript; $('statusSub').textContent = transcript; }; recognition.onerror = () => { recognizing = false; $('statusText').textContent = 'Try again'; }; recognition.onend = () => { recognizing = false; $('micWrap') .classList .remove('listening'); $('micBtn') .classList .remove('stop-mode'); if ( finalTranscript ) { processTranslation( finalTranscript ); } }; recognition.start(); recognizing = true; $('micWrap') .classList .add('listening'); $('micBtn') .classList .add('stop-mode'); $('statusText').textContent = 'Listening...'; $('statusSub').textContent = ''; }
